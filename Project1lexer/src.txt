
#include <stdio.h>
#include <math.h>
#include <Windows.h>

//(x^2 + y^2 - 1)^3 - x^2*y^3 = 0

int main() {
	for (int y = 15; y >= -1; y = y - 1) {
		for (int x = -10; x <= 10; x = x + 5) {
			int a = x * x + y * y - 1;
			putchar(pow(a, 3) - pow(x, 2)*pow(y, 3));
		}
		putchar('\n');
	}
	system("pause");
}
